import sounddevice as sd
import vosk, json, sys
vosk.SetLogLevel(-1)
model = vosk.Model('vosk-model-small-en-us-0.15')
rec = vosk.KaldiRecognizer(model, 16000)
print('Speak now... (say check this, your name, yes, stop etc)')
print('Press Ctrl+C to stop')
with sd.RawInputStream(samplerate=16000, blocksize=2400, dtype='int16', channels=1) as s:
    while True:
        data, _ = s.read(2400)
        if rec.AcceptWaveform(bytes(data)):
            r = json.loads(rec.Result())
            if r.get('text'):
                print('HEARD:', r['text'])
